def Read_txt(file, separator):
    if separator:
        separator = separator
    else:
        separator = '\n'
    with open(file, 'r', encoding='utf-8') as f:
        content = f.read().split(separator)
    print(f'Содержимое текстового документа: {content}')


file = input('Введите директорию .txt файла: ')
separator = input('По какому разделителю будет разделён текст? ')
Read_txt(file, separator)

def Work4():
    troithniki = 0
    direcroria = input('Введите директиве txt файла: ')
    with open(direcroria, 'r', encoding='utf-8') as f:
        sudent = f.read().splitlines()
    print(sudent)
    for i in sudent:
        rob = int(i[-1])
        troithniki += rob
        if rob < 3:
            print(i)
    troithniki /= len(sudent)
    print(f'Средний балл класса:{troithniki}')
